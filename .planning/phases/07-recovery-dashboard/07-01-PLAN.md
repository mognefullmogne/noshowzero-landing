export interface ActiveOffer {

  readonly offered_at: string;
  readonly expires_at: string;

}

   - Clean up interval on unmount.

   - Slot time: formatted from `original_appointment.scheduled_at` using `it-IT` locale (day + time)
   - Countdown badge: amber background when >15 min, red background when <=15 min, gray when 0

6. Header: Use Zap icon with "Offerte Attive" label (consistent with existing section headers).

7. Style: Match existing dashboard card style -- `rounded-2xl border border-black/[0.04] bg-white p-5 shadow-sm`.
