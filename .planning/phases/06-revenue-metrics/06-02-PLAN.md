               const data = await res.json();
               if (data.success) {
                 clinicForm.reset({ avg_appointment_value: data.data.avg_appointment_value });

         setClinicSaved(false);
         const res = await fetch("/api/settings/tenant", {
           method: "PATCH",
           headers: { "Content-Type": "application/json" },
           body: JSON.stringify({ avg_appointment_value: values.avg_appointment_value }),
         });
         const data = await res.json();
         if (!res.ok || !data.success) {
           setClinicError(data.error?.message ?? "Errore nel salvataggio");
           return;
         }
         setClinicSaved(true);
         setTimeout(() => setClinicSaved(false), 3000);
       }
